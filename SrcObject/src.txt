#include "stdio.h"
#include "stdlib.h"
#include "wait.h"
#include "unistd.h"
#include "sys/stat.h"
#include "sys/wait.h"
#include "fcntl.h"
#include "syslog.h"
#include "time.h"
#include "string.h"

int main(int argc, char *argv[])
{
	pid_t pid;
	int fd[2];
	char acBuf[2048];
	
	if(pipe(fd) < 0)
	{
		perror("pipe error");
		exit(1);
	}

	pid = fork();
	if(pid < 0)
	{
		perror("fork error");
		exit(1);
	}

	if(0 == pid)
	{
		close(fd[0]);//close read
		
	}else{
		close(fd[1]);//close write

	
		if( pid != wait(NULL) )
		{
			perror("wait error");
			exit(1);
		}
	}
	
	return 0;
}
